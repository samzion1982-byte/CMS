STEP 4: Integrate into React Component
import tamilMathai from './data/Mathai_01.json'

const getVerse = (book, chapter, verse) => {
  return tamilMathai.find(v => 
    v.chapter === chapter && v.verse === verse
  )
}

STEP 5: Display Verses
const verse = getVerse('Mathai', 1, 1)
console.log(verse.text) // Display Tamil text
